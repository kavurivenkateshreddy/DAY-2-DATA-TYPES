'''type() > this method will help us to know what datatype it is.'''
'''string > anything encloed between single quotes or double quotes
or three times single quotes or three times double quotes
kept inside the print statement or assigned to a variable is called as string.'''
'''boolean > this method is used to determined the statement is true or falsr'''
print(type(1))        #'int'
print(type(-2))       #'int'
print(type(0.1))      #'float'
print(type(-0.2))     #'float'
print(type('python')) #'str'
print(type("python")) #'str'
print(type('''python''')) #'str'
print(type("""python""")) #'str'
print(type(3<5))      #'bool'
print(type(5<3))      #'bool'
print(type(True))     #'bool'
print(type(False))    #'bool'
